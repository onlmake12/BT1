### Title
Verification Fees Permanently Locked in `PythLazer.sol` With No Withdrawal Mechanism - (File: `lazer/contracts/evm/src/PythLazer.sol`)

---

### Summary

`PythLazer.sol` collects a `verification_fee` on every call to `verifyUpdate`, but the contract contains no function to withdraw or recover those accumulated fees. Every wei paid as a verification fee is permanently locked in the contract under the current implementation.

---

### Finding Description

`PythLazer.sol` declares a public state variable `verification_fee` initialized to `1 wei` in `initialize()`. [1](#0-0) 

The `verifyUpdate` function is `payable`, enforces `msg.value >= verification_fee`, and refunds any excess above the fee back to the caller: [2](#0-1) 

After the refund, exactly `verification_fee` wei remains in the contract per call. Scanning the entire contract reveals:

- No `withdrawFees`, `withdraw`, or any ETH-extraction function.
- No `setVerificationFee` setter — the fee is frozen at 1 wei forever under this implementation.
- No `receive()` or `fallback()` function.
- The only functions present are `initialize`, `_authorizeUpgrade`, `updateTrustedSigner`, `isValidSigner`, `verifyUpdate`, and `version`. [3](#0-2) 

The contract inherits `UUPSUpgradeable` and `OwnableUpgradeable`, so an upgrade could theoretically add a withdrawal function, but the current deployed implementation provides no such path.

---

### Impact Explanation

All ETH paid as verification fees accumulates in the `PythLazer` contract and is irrecoverable under the current implementation. Protocol revenue from Lazer price-feed verification is permanently locked. As Lazer adoption grows and call volume increases, the locked balance grows proportionally. Additionally, because there is no `setVerificationFee` function, the protocol cannot raise the fee to generate meaningful revenue without deploying an upgrade — and even after an upgrade, previously locked fees remain inaccessible unless the upgrade explicitly adds a withdrawal path.

---

### Likelihood Explanation

Every unprivileged Lazer updater or relayer that calls `verifyUpdate` with a valid signed update triggers the fee lock. This is the primary entry point for all Lazer price-feed consumers on EVM chains. The likelihood of fee accumulation is **certain** given any real usage of the Lazer product.

---

### Recommendation

1. Add a `withdrawFees(address recipient, uint256 amount) external onlyOwner` function to allow the contract owner to recover accumulated verification fees.
2. Add a `setVerificationFee(uint256 newFee) external onlyOwner` function so the fee can be adjusted without requiring a full contract upgrade.

Example:
```solidity
function withdrawFees(address payable recipient, uint256 amount) external onlyOwner {
    require(address(this).balance >= amount, "Insufficient balance");
    recipient.transfer(amount);
}

function setVerificationFee(uint256 newFee) external onlyOwner {
    verification_fee = newFee;
}
```

---

### Proof of Concept

1. Deploy `PythLazer` (or use the existing proxy).
2. Call `verifyUpdate{value: 1 wei}(validUpdate)` — the call succeeds, 1 wei stays in the contract.
3. Repeat N times. `address(pythLazer).balance == N wei`.
4. Attempt to call any function to withdraw — none exists. The ETH is permanently locked.
5. Confirm: `address(pythLazer).balance` is non-zero and no function in the ABI can reduce it. [4](#0-3)

pragma solidity ^0.8.13;

import "@openzeppelin/contracts-upgradeable/proxy/utils/UUPSUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/access/OwnableUpgradeable.sol";
import "@openzeppelin/contracts/utils/cryptography/ECDSA.sol";

contract PythLazer is OwnableUpgradeable, UUPSUpgradeable {
    TrustedSignerInfo[100] internal trustedSigners;
    uint256 public verification_fee;
    mapping(address => uint256) trustedSignerToExpiresAtMapping;

    constructor() {
        _disableInitializers();
    }

    struct TrustedSignerInfo {
        address pubkey;
        uint256 expiresAt;
    }

    function initialize(address _topAuthority) public initializer {
        __Ownable_init(_topAuthority);
        __UUPSUpgradeable_init();

        verification_fee = 1 wei;
    }

    function _authorizeUpgrade(address) internal override onlyOwner {}

    function updateTrustedSigner(
        address trustedSigner,
        uint256 expiresAt
    ) external onlyOwner {
        if (expiresAt == 0) {
            for (uint8 i = 0; i < trustedSigners.length; i++) {
                if (trustedSigners[i].pubkey == trustedSigner) {
                    trustedSigners[i].pubkey = address(0);
                    trustedSigners[i].expiresAt = 0;
                    delete trustedSignerToExpiresAtMapping[trustedSigner];
                    return;
                }
            }
            revert("no such pubkey");
        } else {
            for (uint8 i = 0; i < trustedSigners.length; i++) {
                if (trustedSigners[i].pubkey == trustedSigner) {
                    trustedSigners[i].expiresAt = expiresAt;
                    trustedSignerToExpiresAtMapping[trustedSigner] = expiresAt;
                    return;
                }
            }
            // Signer not found - adding a new signer.
            for (uint8 i = 0; i < trustedSigners.length; i++) {
                if (trustedSigners[i].pubkey == address(0)) {
                    trustedSigners[i].pubkey = trustedSigner;
                    trustedSigners[i].expiresAt = expiresAt;
                    trustedSignerToExpiresAtMapping[trustedSigner] = expiresAt;
                    return;
                }
            }
            revert("no space for new signer");
        }
    }

    function isValidSigner(address signer) public view returns (bool) {
        return block.timestamp < trustedSignerToExpiresAtMapping[signer];
    }

    function verifyUpdate(
        bytes calldata update
    ) external payable returns (bytes calldata payload, address signer) {
        // Require fee and refund excess
        require(msg.value >= verification_fee, "Insufficient fee provided");
        if (msg.value > verification_fee) {
            payable(msg.sender).transfer(msg.value - verification_fee);
        }

        if (update.length < 71) {
            revert("input too short");
        }
        uint32 EVM_FORMAT_MAGIC = 706910618;

        uint32 evm_magic = uint32(bytes4(update[0:4]));
        if (evm_magic != EVM_FORMAT_MAGIC) {
            revert("invalid evm magic");
        }
        uint16 payload_len = uint16(bytes2(update[69:71]));
        if (update.length < 71 + payload_len) {
            revert("input too short");
        }
        payload = update[71:71 + payload_len];
        bytes32 hash = keccak256(payload);
        (signer, , ) = ECDSA.tryRecover(
            hash,
            uint8(update[68]) + 27,
            bytes32(update[4:36]),
            bytes32(update[36:68])
        );
        if (signer == address(0)) {
            revert("invalid signature");
        }
        if (!isValidSigner(signer)) {
            revert("invalid signer");
        }
    }

    function version() public pure returns (string memory) {
        return "0.1.1";
    }
}
